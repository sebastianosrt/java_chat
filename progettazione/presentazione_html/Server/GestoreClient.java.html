6    import java.io.*;
7    import java.net.Socket;

10   /**
11    * Questa classe riceve i messaggi da un client ed esegue un comando dato
12    *
13    * @author Sebastiano Sartor
14    */
15   class GestoreClient implements Runnable {
16       private final Server server;
17       private final Socket client;
18       private PrintWriter output;
19       private BufferedReader input;
20       private String username = null;
21   
22       public GestoreClient(Server server, Socket client) {
23           this.server = server;
24           this.client = client;
25       }
26   
27       @Override
28       public void run() {
29           try {
30               init();
31               while (client.isConnected()) {
32                   // riceve il messaggio che sar√† una JSON string
33                   String messaggio = input.readLine();
34                   if (messaggio != null) {
35                       JSONObject req = new JSONObject(messaggio);
36                       JSONObject res = null;
37                       String comando = req.getString("comando");
38   
39                       switch (comando) {
40                           case "invia_messaggio":
41                               res = MySQL.addMessage(req.getString("sorgente"), req.getString("destinatario"), req.getString("type"), req.getString("data"));
42                               if (res.getString("risultato").equals("true")) {
43                                   output.println(res);
44                                   req.put("id", res.getInt("inserted_id"));
45                                   // converto la stringa in oggetto e prendo il valore del campo destinatario
46                                   String destinatario = req.getString("destinatario");
47                                   // prende i client connessi
48                                   ArrayList<GestoreClient> clients = server.getClients();
49                                   // ricerca del destinatario tra i client ed invia il messaggio
50                                   boolean sent = false;
51                                   for (GestoreClient c : clients) {
52                                       if (c.getUsername() != null && c.getUsername().equals(destinatario)) {
53                                           c.inviaMessaggio(req.toString());
54                                           sent = true;
55                                       }
56                                   }
57                                   if (!sent)
58                                       MySQL.addContact(req.getString("destinatario"), req.getString("sorgente"));
59                               }
60                               break;
61                           case "invia_file": {
62                               String mittente = req.getString("sorgente");
63                               String destinatario = req.getString("destinatario");
64   
65                               req = new JSONObject(input.readLine());
66                               int size = req.getInt("size");
67                               byte[] buffer = new byte[size]; // or 4096, or more
68   
69                               String filename = req.getString("file_name");
70   
71                               String path = new File(".").getCanonicalPath();
72                               path += "\\Files\\";
73   
74                               InputStream is = client.getInputStream();
75                               String finalPath = path;
76                               // thread che legge dal socket e scrive il file
77                               new Thread(() -> {
78                                   try {
79                                       FileOutputStream fo = new FileOutputStream(finalPath + filename);
80                                       int count;
81                                       while (fo.getChannel().size() < size - 1 && (count = is.read(buffer)) > 0)
82                                           fo.write(buffer, 0, count);
83                                   } catch (IOException e) {
84                                       System.out.println("non risesco a scrivere il file");
85                                   }
86                               }).start();
87   
88                               int id = 0;
89   
90                               res = MySQL.addMessage(mittente, destinatario, "file", filename);
91                               output.println(res);
92   
93                               boolean sent = false;
94                               ArrayList<GestoreClient> clients = server.getClients();
95                               req = new JSONObject();
96                               req.put("sorgente", mittente);
97                               req.put("destinatario", destinatario);
98                               req.put("comando", "invia_file");
99                               req.put("file_name", filename);
100                              req.put("id", id);
101                              for (GestoreClient c : clients) {
102                                  if (c.getUsername() != null && c.getUsername().equals(destinatario)) {
103                                      c.inviaMessaggio(req.toString());
104                                      sent = true;
105                                  }
106                              }
107                              if (!sent)
108                                  MySQL.addContact(destinatario, mittente);
109                              break;
110                          }
111                          case "get_file": {
112                              String fileName = req.getString("file_name");
113                              // leggo file e prendo la dimensione
114                              FileInputStream fs = new FileInputStream("./Files/" + fileName);
115                              int size = (int) fs.getChannel().size();
116                              // invio dimensione file e nome file
117                              res = new JSONObject();
118                              res.put("size", size);
119                              output.println(res);
120                              // invio i bytes
121                              byte[] b = new byte[size];
122                              fs.read(b, 0, b.length);
123                              OutputStream os = client.getOutputStream();
124                              os.write(b, 0, b.length);
125                              os.close();
126                              break;
127                          }
128                          case "set_username":
129                              setUsername(req.getString("username"));
130                              break;
131                          case "login":
132                              res = MySQL.authentication(req.getString("username"), req.getString("password"));
133                              break;
134                          case "registrazione":
135                              res = MySQL.addUser(req.getString("username"), req.getString("password"));
136                              break;
137                          case "get_contatti":
138                              res = MySQL.getListContacts(req.getString("sorgente"));
139                              break;
140                          case "add_contatto":
141                              res = MySQL.addContact(req.getString("sorgente"), req.getString("contatto"));
142                              break;
143                          case "cerca_utenti":
144                              res = MySQL.searchUser(req.getString("nome_utente"));
145                              break;
146                          case "get_messaggi":
147                              res = MySQL.getMessaggi(req.getString("sorgente"), req.getString("contatto"));
148                              break;
149                          case "elimina_messaggio":
150                              res = MySQL.deleteMessage(req.getInt("id_messaggio"));
151                              if (res.getString("risultato").equals("true")) {
152                                  String destinatario = req.getString("contatto");
153                                  ArrayList<GestoreClient> clients = server.getClients();
154                                  for (GestoreClient c : clients)
155                                      if (c.getUsername() != null && c.getUsername().equals(destinatario))
156                                          c.inviaMessaggio(req.toString());
157                              }
158                              break;
159                          case "disconnect":
160                              this.destroy();
161                              break;
162                      }
163                      if (res != null)
164                          output.println(res);
165                  }
166              }
167              this.destroy();
168          } catch (IOException e) {
169              // client si disconnette
170              try {
171                  this.destroy();
172              } catch (IOException ex) {
173                  ex.printStackTrace();
174              }
175          }
176      }
177  
178      /**
179       * Inizializza gli stream
180       *
181       * @throws IOException
182       */
183      private void init() throws IOException {
184          output = new PrintWriter(new BufferedOutputStream(client.getOutputStream()), true);
185          input = new BufferedReader(new InputStreamReader(new BufferedInputStream(client.getInputStream())));
186      }
187  
188      /**
189       * Chiude gli stream ed i socket e rimuove "this" dalla lista dei gestore client del server
190       *
191       * @throws IOException
192       */
193      private void destroy() throws IOException {
194          server.rimuoviClient(this);
195          output.close();
196          input.close();
197          client.close();
198      }

206          output.println(messaggio);
207      }

224          this.username = username;
225      }
