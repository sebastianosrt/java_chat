1    package Server;
2    
3    import Server.MySQL.MySQL;
4    
5    import java.io.IOException;
6    import java.net.ServerSocket;
7    import java.net.Socket;
8    import java.util.ArrayList;
9    
10   /**
11    * Apre il server, si connette al database ed accetta le connessioni
12    *
13    * @author Sebastiano Sartor
14    */
15   class Server implements Runnable {
16       private final ArrayList<GestoreClient> clients = new ArrayList<>();
17   
18       @Override
19       public void run() {
20           try {
21               MySQL.openConnection();
22               // creo il server socket
23               ServerSocket server = new ServerSocket(666);
24               // ascolta all'infinito richieste di connessione
25               while (!server.isClosed()) {
26                   // ricevi connessione da un client
27                   Socket socket = server.accept();
28                   // quando riceve una connessione aggiunge un nuovo client alla lista e crea un nuovo gestore client
29                   GestoreClient client = new GestoreClient(this, socket);
30                   clients.add(client);
31                   new Thread(client).start();
32               }
33           } catch (IOException e) {
34               System.out.println();
35           }
36           MySQL.closeConnection();
37       }
38   
39       /**
40        * Ritorna i client connessi
41        *
42        * @return clients - lista di gestore client
43        */
44       public ArrayList<GestoreClient> getClients() {
45           return clients;
46       }
47   
48       /**
49        * rimuove un gestore client dalla lista
50        *
51        * @param client - il gestore client da rimuovere
52        */
53       public void rimuoviClient(GestoreClient client) {
54           clients.remove(client);
55       }
56   }
