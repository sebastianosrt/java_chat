138      /**
139       * Questo metodo setta gli event listeners
140       */
141      private void setEventListeners() {
142          // invia messaggio
143          sendBtn.setOnMouseClicked(e -> {
144              addMessaggio(inviaMessaggio(username, message_f.getText()));
145              message_f.setText("");
146          });
147          // quando viene premuto enter invia un messaggio
148          body.setOnKeyReleased(e -> {
149              if (e.getCode() == KeyCode.ENTER) {
150                  addMessaggio(inviaMessaggio(username, message_f.getText()));
151                  message_f.setText("");
152              }
153          });
154          // chiudi scheda
155          closeBtn.setOnMouseClicked(e -> {
156              client.exit();
157              Platform.exit();
158          });
159          // minimizza scheda
160          minimizeBtn.setOnMouseClicked(e -> ((Stage) ((ImageView) e.getSource()).getScene().getWindow()).setIconified(true));
161          // ricerca utente
162          search_f.setOnKeyReleased(e -> {
163              if (search_f.getText().length() > 0) caricaContatti(client.searchUsers(search_f.getText()));
164              else caricaContatti(client.getContattiFromDataBase());
165          });
166          //logout
167          logOut.setOnMouseClicked(e -> {
168              client.destroy();
169              apriLogInView(e);
170          });
171          // muove la finestra quando viene trascinata
172          body.setOnMousePressed(e -> {
173              xOffset = e.getSceneX();
174              yOffset = e.getSceneY();
175          });
176          body.setOnMouseDragged(e -> {
177              stage.setX(e.getScreenX() - xOffset);
178              stage.setY(e.getScreenY() - yOffset);
179          });
180          // menu
181          elimina.setOnAction(event -> {
182              String id = selectedMessage.getId();
183              client.eliminaMessaggio(Integer.parseInt(id), contattoAttivo);
184              chatContaier.getChildren().remove(selectedMessage);
185          });
186          copia.setOnAction(event -> {
187              TextFlow f = (TextFlow) selectedMessage.getChildren().get(0);
188              String str = f.getAccessibleText();
189              // copy on clipboard
190              Toolkit.getDefaultToolkit().getSystemClipboard().setContents(new StringSelection(str), null);
191          });
192          // files
193          fileBtn.setOnMouseClicked(event -> {
194              FileChooser fc = new FileChooser();
195              File selected = fc.showOpenDialog(null);
196              if (selected != null) {
197                  int id = client.inviaFile(contattoAttivo, selected.getAbsolutePath(), selected.getName());
198                  addFile(this.username, selected.getName(), id);
199              }
200          });
201      }

212      /**
213       * Questo metodo invia il messaggio al destinatario
214       *
215       * @param username - l'username del mittente del messaggio
216       * @param message  - il contenuto del messaggio
217       * @return il messaggio da che sarà aggiunto alla view
218       */
219      public Messaggio inviaMessaggio(String username, String message) {
220          Messaggio m = new Messaggio(0, message, username, contattoAttivo, "text");
221          if (username.equals(contattoAttivo) || username.equals(this.username)) {
222              if (message.length() > 0) {
223                  //toglie gli \n finali
224                  while (message.length() > 0 && message.charAt(message.length() - 1) == '\n')
225                      message = message.substring(0, message.length() - 1);
226                  if (message.length() > 0) {
227                      if (newContact) {
228                          client.addContactToDataBase(contattoAttivo);
229                          newContact = false;
230                          caricaContatti(client.getContattiFromDataBase());
231                      }
232                      raiseContact();
233                      m.id = client.inviaMessaggio(this.contattoAttivo, message);
234                      return m;
235                  }
236              }
237          }
238          return null;
239      }

241      /**
242       * Questo metodo aggiunge alla view un nuovo messaggio e gli aggiunge un event listener per il tasto destro del mouse
243       *
244       * @param message - il messaggio da aggiungere alla view
245       */
246      public void addMessaggio(Messaggio message) {
247          if (message != null) {
248              if (message.type.equals("text")) {
249                  String username = message.mittente;
250                  String testo = message.testo;
251                  if (username.equals(contattoAttivo) || username.equals(this.username)) {
252                      if (testo.length() > 0) {
253                          //toglie gli \n finali
254                          while (testo.length() > 0 && testo.charAt(testo.length() - 1) == '\n')
255                              testo = testo.substring(0, testo.length() - 1);
256                          if (testo.length() > 0) {
257                              int id = message.id;
258                              Text text = new Text(testo);
259                              if (!this.username.equals(username))
260                                  text.setFill(Color.WHITE);
261                              else
262                                  text.setFill(Color.BLACK);
263                              TextFlow tempFlow = new TextFlow();
264  
265                              tempFlow.getChildren().add(text);
266                              tempFlow.setMaxWidth(200);
267                              TextFlow flow = new TextFlow(tempFlow);
268                              HBox hbox = new HBox(12);
269                              if (!this.username.equals(username)) {
270                                  tempFlow.getStyleClass().add("tempFlowFlipped");
271                                  flow.getStyleClass().add("textFlowFlipped");
272                                  chatContaier.setAlignment(Pos.TOP_LEFT);
273                                  hbox.setAlignment(Pos.CENTER_LEFT);
274                              } else {
275                                  tempFlow.getStyleClass().add("tempFlow");
276                                  flow.getStyleClass().add("textFlow");
277                                  hbox.setAlignment(Pos.BOTTOM_RIGHT);
278                              }
279                              hbox.getChildren().add(flow);
280                              hbox.getStyleClass().add("hbox");
281                              hbox.setId(id + "");
282                              flow.setAccessibleText(testo);
283                              // set evento - seleziona messaggio
284                              hbox.setOnMouseClicked((MouseEvent e) -> {
285                                  if (e.getButton() == MouseButton.SECONDARY) {
286                                      selectedMessage = (HBox) e.getSource();
287                                      menu.show(scrollPane, e.getScreenX(), e.getScreenY());
288                                  }
289                              });
290                              chatContaier.getChildren().addAll(hbox);
291                          }
292                      }
293                  } else {
294                      if (!hasContact(message.mittente)) {
295                          client.addContactToDataBase(message.mittente);
296                          caricaContatti(client.getContattiFromDataBase());
297                      }
298                  }
299              } else {
300                  addFile(message.mittente, message.testo, message.id);
301              }
302          }
303      }

321                          client.getFile(b.getAccessibleText(), path.getAbsolutePath());
322                  });
323                  TextFlow tempFlow = new TextFlow();
324                  tempFlow.getChildren().add(b);
325                  TextFlow flow = new TextFlow(tempFlow);
326                  HBox hbox = new HBox(12);
327                  if (!this.username.equals(username)) {
328                      tempFlow.getStyleClass().add("tempFlowFlipped");
329                      flow.getStyleClass().add("textFlowFlipped");
330                      chatContaier.setAlignment(Pos.TOP_LEFT);
331                      hbox.setAlignment(Pos.CENTER_LEFT);
332                  } else {
333                      tempFlow.getStyleClass().add("tempFlow");
334                      flow.getStyleClass().add("textFlow");
335                      hbox.setAlignment(Pos.BOTTOM_RIGHT);
336                  }
337                  hbox.getChildren().add(flow);
338                  hbox.getStyleClass().add("hbox");
339                  hbox.setId(id + "");
340                  flow.setAccessibleText(filename);
341                  // set evento - seleziona messaggio
342                  hbox.setOnMouseClicked((MouseEvent e) -> {
343                      if (e.getButton() == MouseButton.SECONDARY) {
344                          selectedMessage = (HBox) e.getSource();
345                          menu.show(scrollPane, e.getScreenX(), e.getScreenY());
346                      }
347                  });
348                  chatContaier.getChildren().addAll(hbox);
349              }
350          } else {
351              if (!hasContact(contattoAttivo)) {
352                  client.addContactToDataBase(contattoAttivo);
353                  caricaContatti(client.getContattiFromDataBase());
354              }
355          }

421      /**
422       * Questo metodo seleziona un contatto
423       *
424       * @param e - l'evento del click di tipo MouseEvent
425       */
426      private void selectContact(MouseEvent e) {
427          // prendo il contatto selezionato dall'evento
428          HBox b = (HBox) e.getSource();
429          // se il contatto selezionato non è già attivo
430          if (b != activeContact) {
431              // setta il contatto attivo
432              setActiveContact(b.getAccessibleText());
433              // toglie il pannello che dice che nessuna chat è selezionata
434              coveringPane.toBack();
435              // toglie tutti i messaggi con il contatto precedente
436              chatContaier.getChildren().clear();
437              // setta il nome del destinatario
438              destinatario_f.setText(b.getAccessibleText());
439              contattoAttivo = b.getAccessibleText();
440              // resetta la ricerca
441              if (search_f.getText().length() > 0) {
442                  search_f.setText("");
443                  caricaContatti(client.getContattiFromDataBase());
444              }
445              caricaMessaggi(client.getMessaggiFromDataBase(contattoAttivo));
446          }
447      }

462      }

464      /**
465       * Questo metodo porta in cima alla lista dei contatti l'ultimo con cui si è messaggato se non è già primo
466       */
467      private void raiseContact() {
468          // assicuro il contatto attivo
469          setActiveContact(contattoAttivo);
470          // metto i contatti in una lista
471          List<Node> nodes = new ArrayList<>(contactsContaier.getChildren());
472          // se il contatto è già in cima
473          if (nodes.get(0).equals(activeContact))
474              return;
475          // prendo l'indice del contatto da portare in alto per rimuovere il separator
476          int i = nodes.indexOf(activeContact);
477          // rimuovo separator
478          nodes.remove(i + 1);
479          // rimuovo il contatto da portare in alto
480          nodes.remove(activeContact);
481          // lo porto in alto (lo metto come primo nodo della lista)
482          nodes.add(0, activeContact);
483          // aggiungo un separator in 2 posizione
484          nodes.add(1, new Separator());
485          // rimuovo tutti i contatti e ci metto la nuova lista
486          contactsContaier.getChildren().clear();
487          contactsContaier.getChildren().addAll(nodes);
488      }

490      /**
491       * Questo metodo controlla se l'utente ha un contatto
492       *
493       * @param name - nome del contatto da controllare
494       * @return true se ha il contatto, altrimenti false
495       */
496      private boolean hasContact(String name) {
497          try {
498              for (Object c : contactsContaier.getChildren())
499                  if (((HBox) c).getAccessibleText().equals(name))
500                      return true;
501          } catch (ClassCastException e) {
502              System.out.println();
503          }
504          return false;
505      }

507      /**
508       * Questo metodo ritorna al login
509       *
510       * @param event - l'oggetto dell'evento click
511       */
512      private void apriLogInView(MouseEvent event) {
513          FXMLLoader loader = new FXMLLoader(getClass().getResource("/Client/Views/LogInView.fxml"));
514          loader.setController(new LogInController(stage));
515          try {
516              Parent root = loader.load();
517              Scene scene = new Scene(root);
518              Node node = (Node) event.getSource();
519              Stage stage = (Stage) node.getScene().getWindow();
520              stage.close();
521              stage.setScene(scene);
522              stage.setResizable(false);
523              stage.show();
524              stage.requestFocus();
525              node.setFocusTraversable(true);
526          } catch (IOException e) {
527              e.printStackTrace();
528          }
529      }
