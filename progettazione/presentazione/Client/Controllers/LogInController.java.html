77       /**
78        * Questo metodo prende le credenziali dalla view, esegue la query al database in caso di errore lo mostra, altrimenti apre l'interfaccia della chat
79        *
80        * @param event - l'evento del click di tipo MouseEvent
81        */
82       public void logIn(MouseEvent event) {
83           // prende le credenziali dalle caselle di testo
84           String username = username_f.getText();
85           String password = password_f.getText();
86           String error;
87           if (username.length() == 0) {
88               error_f.setText("Inserisci un'username");
89               return;
90           }
91           if (password.length() == 0) {
92               error_f.setText("Inserisci una password");
93               return;
94           }
95           error = Autenticazione.login(username, password);
96           if (error.length() > 0) {
97               error_f.setText(error);
98               return;
99           }
100          // apre la chat
101          try {
102              apriChatView(username, event);
103          } catch (IOException e) {
104              error_f.setText("Errore interno\nImpossible aprire la chat");
105          }
106      }
107  
108      /**
109       * Questo metodo registra un utente, in caso di errore lo mostra
110       */
111      public void signUp() {
112          response.setText("");
113          // prende le credenziali dalle caselle di testo
114          String username = username_r.getText();
115          String password = password_r.getText();
116          String error;
117          if (username.length() == 0) {
118              error_r.setText("Inserisci un'username");
119              return;
120          }
121          if (password.length() == 0) {
122              error_r.setText("Inserisci una password");
123              return;
124          }
125          error = Autenticazione.signup(username, password);
126          if (error.length() > 0)
127              error_r.setText(error);
128          else {
129              response.setText("Registrato correttamente!");
130              password_r.setText("");
131              username_r.setText("");
132          }
133      }
134  
135      /**
136       * Questo metodo apre la chat
137       *
138       * @param username - l'username dell'utente che ha aperto la chat
139       * @param event    - l'oggetto dell'evento click
140       * @throws IOException
141       */
142      private void apriChatView(String username, MouseEvent event) throws IOException {
143          Node node = (Node) event.getSource();
144          Stage stage = (Stage) node.getScene().getWindow();
145  
146          FXMLLoader loader = new FXMLLoader(getClass().getResource("/Client/Views/ChatView.fxml"));
147          loader.setController(new ClientController(stage));
148          ClientController controller = loader.getController();
149          controller.setUsername(username);
150  
151          Parent root = loader.load();
152          Scene scene = new Scene(root);
153          stage.close();
154          stage.setScene(scene);
155          stage.setResizable(false);
156          stage.show();
157          stage.requestFocus();
158          node.setFocusTraversable(true);
159      }
160  
161      /**
162       * Questo metodo gestisce i click del mouse
163       *
164       * @param event - l'oggetto dell'evento click
165       */
166      @FXML
167      private void handleMouseClick(MouseEvent event) {
168          error_r.setText("");
169          error_f.setText("");
170          if (event.getSource() == submit_f) {
171              logIn(event);
172          } else if (event.getSource() == submit_r) {
173              signUp();
174          } else if (event.getSource() == registrati_label || event.getSource() == registrati_f) {
175              new Pulse(registrazionePanel).play();
176              registrazionePanel.toFront();
177          } else if (event.getSource() == login_label || event.getSource() == login_r) {
178              new Pulse(loginPanel).play();
179              loginPanel.toFront();
180          }
181      }

183      @Override
184      public void initialize(URL location, ResourceBundle resources) {
185          error_f.setText("");
186          error_r.setText("");
187          response.setText("");
188  
189          // chiudi scheda
190          closeBtn.setOnMouseClicked(e -> Platform.exit());
191          closeBtn1.setOnMouseClicked(e -> Platform.exit());
192  
193          // minimizza scheda
194          minimizeBtn.setOnMouseClicked(e -> ((Stage) ((ImageView) e.getSource()).getScene().getWindow()).setIconified(true));
195          minimizeBtn1.setOnMouseClicked(e -> ((Stage) ((ImageView) e.getSource()).getScene().getWindow()).setIconified(true));
196  
197          // muove la finestra quando viene trascinata
198          loginPanel.setOnMousePressed(event12 -> {
199              xOffset = event12.getSceneX();
200              yOffset = event12.getSceneY();
201          });
202          loginPanel.setOnMouseDragged(event1 -> {
203              stage.setX(event1.getScreenX() - xOffset);
204              stage.setY(event1.getScreenY() - yOffset);
205          });
206          registrazionePanel.setOnMousePressed(event12 -> {
207              xOffset = event12.getSceneX();
208              yOffset = event12.getSceneY();
209          });
210          registrazionePanel.setOnMouseDragged(event1 -> {
211              stage.setX(event1.getScreenX() - xOffset);
212              stage.setY(event1.getScreenY() - yOffset);
213          });
214      }
