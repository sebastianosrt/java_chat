1    package Client;

3    import Client.Controllers.LogInController;
4    import javafx.application.Application;
5    import javafx.fxml.FXMLLoader;
6    import javafx.scene.Parent;

8    import javafx.stage.Stage;

10   
11   import java.io.IOException;
12   
13   /**
14    * Apre la pagina di log in
15    *
16    * @author Sebastiano Sartor
17    */
18   public class Main extends Application {
19       public static void main(String[] args) {
20           launch(args);
21       }
22   
23       @Override
24       public void start(Stage primaryStage) throws IOException {
25           // carica il file fxml
26           FXMLLoader loader = new FXMLLoader(getClass().getResource("Views/LogInView.fxml"));
27           // setta il controller della pagina
28           loader.setController(new LogInController(primaryStage));
29           // apre la finestra
30           Parent root = loader.load();
31           Scene scene = new Scene(root);
32           primaryStage.initStyle(StageStyle.UNDECORATED);
33           primaryStage.setResizable(false);
34           primaryStage.setScene(scene);
35           primaryStage.show();
36           primaryStage.requestFocus();
37       }
38   }
