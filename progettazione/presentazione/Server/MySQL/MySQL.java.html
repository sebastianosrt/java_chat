195      /**
196       * Metodo utilile all'interno della classe statica MySQL
197       *
198       * @param rs risultato della query
199       * @return mappa di stringhe (simile ad un array associativo [php])
200       * @throws SQLException in caso di errori nell'esecuzione della query
201       */
202      private static HashMap<String, String> createRowObj(ResultSet rs) throws SQLException {
203          HashMap<String, String> map = new HashMap<>();
204  
205          ResultSetMetaData rsmd = rs.getMetaData();
206          int columnsNumber = rsmd.getColumnCount();
207  
208          for (int i = 1; i <= columnsNumber; i++) {
209              map.put(rsmd.getColumnName(i), rs.getString(i));
210          }
211  
212          return map;
213      }

401              if (!statusConn) {
402

404                  pstmt.setString(1, mittente);
405                  pstmt.setString(2, destinatario);
406                  pstmt.setString(3, type); //default text
